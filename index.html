class Store {
								@observable public data;
								public async loadData() {...}
							}

							class ViewModel {
								private store: Store;
								@observable public loading = false;
								public constructor() {
									loading = true;
									store.loadData();
									loading = false;
								}
							}
							
							@observer
							class Component {
								<ComponentLoader 
									renderChild={...data...}
									isLoaded={loading}
								>
							}
						

						class Loading {
							@observable private data;

							public static ComponentLoader(
								render: (data) => React.Element
							) {
								return {... data ...}
							}

							public merge(
								loadings: []Loading, 
								mergeFunction: (data) => newData
							): Loading {...}
						}
					

						async function merge(promise1: Promise, promise2: Promise) {
							const value1 = await promise1;
							const value2 = await promise2;

							return value1 + value2;
						}

						async function extract(promise: Promise) {
							const data = await promise;
							return data.subdata;
						}
